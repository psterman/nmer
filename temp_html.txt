<script>
(function () {
  var strip = document.getElementById("strip");
  var favicon = document.getElementById("favicon");
  var buttons = document.querySelectorAll(".btn");
  var btnSearch = document.getElementById("btn-search");
  var selectedAction = "";

  function postToAhk(obj) {
    try {
      if (window.chrome && window.chrome.webview)
        window.chrome.webview.postMessage(obj);
    } catch (e) {}
  }

  function setSelected(action) {
    selectedAction = action;
    buttons.forEach(function (b) {
      b.classList.toggle("selected", b.dataset.action === action);
    });
  }

  function setSearchPulse(on) {
    btnSearch.classList.toggle("pulse", !!on);
  }

  function setSearchDragOver(on) {
    btnSearch.classList.toggle("drag-over", !!on);
  }

  function setSearchLoading(on) {
    btnSearch.classList.toggle("loading", !!on);
  }

  // жҢүй’®зӮ№еҮ»
  buttons.forEach(function (btn) {
    btn.addEventListener("click", function (e) {
      e.stopPropagation();
      var action = btn.dataset.action;
      setSelected(action);
      if (action === "Search") {
        postToAhk({ type: "toolbar_search_click" });
      } else {
        postToAhk({ type: "toolbar_action", action: action });
      }
    });
  });

  // з©әзҷҪеҢәжӢ–еҠЁж•ҙзӘ?  strip.addEventListener("mousedown", function (e) {
    if (e.target === strip || e.target === favicon || e.target.parentElement === favicon) {
      postToAhk({ type: "drag_host" });
    }
  });

  // ж»ҡиҪ®зј©ж”ҫ
  strip.addEventListener("wheel", function (e) {
    e.preventDefault();
    postToAhk({ type: "wheel", delta: e.deltaY > 0 ? -1 : 1 });
  }, { passive: false });

  // еҸій”®иҸңеҚ•
  strip.addEventListener("contextmenu", function (e) {
    e.preventDefault();
    postToAhk({ type: "context_menu", x: e.screenX, y: e.screenY });
  });

  // жӢ–ж”ҫпјҡжҗңзҙўжҢүй’?  async function readDropTextAsync(dt) {
    if (!dt) return "";
    try {
      if (dt.items && dt.items.length) {
        for (var i = 0; i < dt.items.length; i++) {
          var it = dt.items[i];
          if (it.kind !== "string") continue;
          var s = await new Promise(function (res) {
            try {
              it.getAsString(function (x) { res(x || ""); });
            } catch (e2) {
              res("");
            }
          });
          if (s && String(s).trim()) return String(s).trim();
        }
      }
      return (dt.getData("text/plain") || dt.getData("Text") || dt.getData("text/uri-list") || "").trim();
    } catch (e) {
      return "";
    }
  }

  async function onDrop(e) {
    e.preventDefault();
    e.stopPropagation();
    setSearchDragOver(false);
    var t = await readDropTextAsync(e.dataTransfer);
    t = (t || "").trim();
    if (!t && e.dataTransfer && e.dataTransfer.files && e.dataTransfer.files.length)
      t = (e.dataTransfer.files[0].name || "").trim();
    if (!t) return;
    setSearchLoading(true);
    postToAhk({ type: "drop_search", text: t });
  }

  function onDragOver(e) {
    e.preventDefault();
    e.stopPropagation();
    try {
      if (e.dataTransfer) e.dataTransfer.dropEffect = "copy";
    } catch (e2) {}
    setSearchDragOver(true);
  }

  function onDragLeave(e) {
    if (!btnSearch.contains(e.relatedTarget))
      setSearchDragOver(false);
  }

  btnSearch.addEventListener("dragenter", function (e) { e.preventDefault(); });
  btnSearch.addEventListener("dragover", onDragOver);
  btnSearch.addEventListener("dragleave", onDragLeave);
  btnSearch.addEventListener("drop", onDrop);

  // е…ЁеұҖжӢ–ж”ҫе…ңеә•
  document.body.addEventListener("dragover", function (e) { e.preventDefault(); });
  document.body.addEventListener("drop", function (e) {
    e.preventDefault();
    if (e.target === btnSearch || btnSearch.contains(e.target)) return;
    onDrop(e);
  });

  // жҺҘж”¶ AHK ж¶ҲжҒҜ
  if (window.chrome && window.chrome.webview) {
    window.chrome.webview.addEventListener("message", function (ev) {
      var d = ev.data;
      if (!d || !d.type) return;
      if (d.type === "SELECTION_CHANGE") {
        setSearchPulse(true);
        return;
      }
      if (d.type === "SELECTION_CLEAR") {
        setSearchPulse(false);
        return;
      }
      if (d.type === "drop_done") {
        setSearchLoading(false);
        setSearchPulse(false);
        return;
      }
      // зј©ж”ҫз”?AHK ж”№е®ҝдё»зӘ—еҸЈе®ҢжҲҗпјҢеӢҝеҜ№ body еҶ?transform
      if (d.type === "set_scale") {
        return;
      }
      if (d.type === "set_logo") {
        var u = d.url || "";
        var im = document.getElementById("favicon-img");
        if (im && u) im.src = u;
        return;
      }
      if (d.type === "set_selected") {
        setSelected(d.action || "");
        return;
      }
    });
  }

  function revealStripOnce() {
    if (revealStripOnce._done) return;
    revealStripOnce._done = true;
    strip.classList.add("strip-visible");
  }

  if (document.readyState === "complete")
    revealStripOnce();
  else
    window.addEventListener("load", revealStripOnce, { once: true });
  setTimeout(revealStripOnce, 2200);

  postToAhk({ type: "toolbar_ready" });
})();
</script>
