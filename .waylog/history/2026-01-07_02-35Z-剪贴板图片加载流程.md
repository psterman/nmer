ShowImagePreviewImmediately(imagePath) {

        ; 获取图片尺寸

        previewWidth := 350

GetImagePathFromRowData(rowData) {
    path := ""
    ; 1. 检查 ImagePath 字段
    if (rowData.Has("ImagePath") && rowData["ImagePath"] != "")
        path := rowData["ImagePath"]
    ; 2. 检查 Content 是否为合法路径
    else if (rowData.Has("Content") && rowData["Content"] != "")
        path := rowData["Content"]
    
    if (path != "") {
        ; 清理可能的双斜杠转义 [cite: 134]
        path := StrReplace(path, "\\", "\")
        path := StrReplace(path, "\", "\")
        
        ; 验证文件后缀
        SplitPath(path, , , &ext)
        if (ext ~= "i)^(png|jpg|jpeg|gif|bmp|webp|ico)$" && FileExist(path))
            return path
    }
    return ""
}

. 预览失败的核心原因分析
坐标变量未定义 (最直接原因)：在你的 ShowHistoryImagePreview 函数中，你使用了 HistoryImagePreviewGUI.Show("x" . x . " y" . y ...)。但在函数内部，变量 x 和 y 根本没有被赋值。AHK v2 会因为变量未初始化而直接终止该函数的执行。

路径解析失败：在 OnHistoryItemSelected 中，你通过 itemData.Content 获取路径。如果这一行是纯图片数据（非文件路径），Content 存储的是图片的二进制摘要或描述，而非路径。必须检查 itemData.ImagePath。

窗口重复销毁逻辑异常：在 ShowHistoryImagePreview 开头，你使用了 if (HistoryImagePreviewGUI != 0) HistoryImagePreviewGUI.Destroy()。如果此时窗口正在显示，频繁的 Destroy 会导致严重的闪烁甚至崩溃。

; ===================== 选中的处理函数 =====================
OnHistoryItemSelected(LVObj, itemIndex, selected) {
    global HistoryImagePreviewTimer, HistoryDisplayCache
    
    ; 1. 只要选中状态改变，就清除旧定时器
    if (HistoryImagePreviewTimer != 0) {
        try SetTimer(HistoryImagePreviewTimer, 0)
        HistoryImagePreviewTimer := 0
    }

    if (!selected) {
        HideHistoryImagePreview()
        return
    }

    ; 2. 获取数据
    if (itemIndex > HistoryDisplayCache.Length) return
    itemData := HistoryDisplayCache[itemIndex]
    
    ; 3. 提取路径：优先使用 ImagePath，其次 Content
    targetPath := ""
    if (itemData.HasProp("ImagePath") && itemData.ImagePath != "")
        targetPath := itemData.ImagePath
    else if (RegExMatch(itemData.Content, "i)\.(png|jpg|jpeg|gif|bmp|webp)$"))
        targetPath := itemData.Content

    ; 4. 执行预览
    if (targetPath != "" && FileExist(targetPath)) {
        ; 设置 150ms 延迟显示，防止快速滚动时卡顿
        HistoryImagePreviewTimer := () => ShowHistoryImagePreview(targetPath)
        SetTimer(HistoryImagePreviewTimer, -150)
    } else {
        HideHistoryImagePreview()
    }
}

; ===================== 显示预览窗口 (修复坐标) =====================
ShowHistoryImagePreview(imagePath) {
    global HistoryImagePreviewGUI, GuiID_ClipboardHistory, HistoryColors, HistoryImagePreviewIsVisible
    
    try {
        ; 1. 获取图片尺寸
        conf := ImagePut.GetConfig(imagePath)
        pw := 350 ; 预设预览宽度
        ph := (conf.height / conf.width) * pw
        
        ; 2. 获取主窗口位置 (计算坐标的核心)
        if !GuiID_ClipboardHistory
            return
            
        WinGetPos(&mX, &mY, &mW, &mH, GuiID_ClipboardHistory.Hwnd)
        
        ; 计算预览窗位置：主窗口右侧 10px
        targetX := mX + mW + 10
        targetY := mY
        
        ; 屏幕边缘检测：如果右边放不下，移到左边
        if (targetX + pw > A_ScreenWidth)
            targetX := mX - pw - 10

        ; 3. 创建/更新窗口
        if (HistoryImagePreviewGUI != 0) {
            try HistoryImagePreviewGUI.Destroy()
        }
            
        HistoryImagePreviewGUI := Gui("+AlwaysOnTop -Caption +ToolWindow +Border")
        HistoryImagePreviewGUI.BackColor := HistoryColors.Background
        
        ; 4. 添加图片
        HistoryImagePreviewGUI.Add("Picture", "x0 y0 w" . pw . " h" . ph, imagePath)
        
        ; 5. 显示 (使用 NoActivate 防止抢占搜索框焦点)
        HistoryImagePreviewGUI.Show("x" . targetX . " y" . targetY . " w" . pw . " h" . ph " NoActivate")
        HistoryImagePreviewIsVisible := true
        
    } catch as err {
        HistoryImagePreviewIsVisible := false
    }
}
