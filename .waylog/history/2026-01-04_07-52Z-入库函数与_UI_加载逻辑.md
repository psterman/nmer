    global ClipboardDB, CursorPanelResultLV, ClipboardCurrentCategory
    if (!ClipboardDB) return
    
    ; 1. 构造过滤条件
    filter := ""
    if (ClipboardCurrentCategory != "" && ClipboardCurrentCategory != "全部") {
        filter := " WHERE DataType = '" ClipboardCurrentCategory "'"
    }
    
    ; 2. 预先从硬盘读取数据到内存变量，不要在 Add 循环里读硬盘
    sql := "SELECT * FROM ClipboardHistory" filter " ORDER BY Timestamp DESC LIMIT 100"
    data := ClipboardDB.GetTable(sql)
    
    ; 3. 锁定界面更新，防止闪烁
    CursorPanelResultLV.Opt("-Redraw") 
    CursorPanelResultLV.Delete()

    }

}
