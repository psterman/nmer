## 2. 增强型采集逻辑 (适配 AHK v2 & WinClip)
重写 OnClipboardChange(Type) 回调函数，实现以下精细化探测流：

A. 来源追踪 (Context Capture)
利用 WinGetID("A") 锁定当前窗口。

调用 WinGetProcessName、WinGetTitle 和 WinGetProcessPath 捕获详尽的来源信息。

B. 智能类型探测器 (Type Detectors)
如果 Type = 1（文本），按优先级执行正则匹配并打标：

Link: i)^(https?:\/\/|www\.)[^\s/$.?#].[^\s]*$

Email: i)^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}$

Code: 识别关键符号组合：[\{\}\[\]\(\);]|=>|:=|function|const|var|let|import。

Text: 以上皆不匹配时的默认值。

统计: 调用 StrLen 计算字符数；通过正则分割计算单词 / 文字数。

富文本: 调用 WinClip.GetHTML()，若存在则在 MetaData 记录 hasHTML: 1。

C. 非文本探测 (Non-Text Capture)
如果 Type = 2：

File: 使用 WinClip.GetFiles() 判断。若是文件，DataType 设为 File，Content 存入完整路径列表。

Image: DataType 设为 Image。注意：此阶段仅在 Content 记录占位符 [Pending_Image_Save]，下一阶段再集成路径保存。

## 3. 安全性与性能 (Performance & Security)

```autohotkey
; ===================== 剪贴板数据库初始化函数（重构版） =====================
InitClipboardDB() {
    global ClipboardDB, ClipboardDBPath
    
    ; 设置数据库文件路径（存储在Data目录）
    DataDir := A_ScriptDir "\Data"
    ClipboardDBPath := DataDir "\CursorData.db"
    
    ; 检查并创建Data目录
    if (!DirExist(DataDir)) {
        try {
            DirCreate(DataDir)
        } catch as e {
            MsgBox("无法创建Data目录: " . e.Message, "数据库初始化错误", "IconX")
            ClipboardDB := 0
            return
        }
    }
    
    ; 检查 sqlite3.dll 是否存在
    DllPath := A_ScriptDir "\sqlite3.dll"
    if (!FileExist(DllPath)) {
        MsgBox("sqlite3.dll 未找到。`n请确保 sqlite3.dll 与脚本位于同一目录。", "数据库初始化错误", "IconX")
        ClipboardDB := 0
        return
    }
    
    ; 创建 SQLiteDB 实例并打开数据库
    try {
        ClipboardDB := SQLiteDB()
        if (!ClipboardDB.OpenDB(ClipboardDBPath)) {
            MsgBox("无法打开数据库: " . ClipboardDBPath . "`n错误: " . ClipboardDB.ErrorMsg, "数据库初始化错误", "IconX")
            ClipboardDB := 0
            return
        }
        
        ; 清理旧表
        SQL := "DROP TABLE IF EXISTS ClipboardHistory"
        if (!ClipboardDB.Exec(SQL)) {
            MsgBox("删除旧表失败: " . ClipboardDB.ErrorMsg, "数据库初始化错误", "IconX")
            ClipboardDB.CloseDB()
            ClipboardDB := 0
            return
        }
        
        ; 创建新表（包含所有必需字段）
        SQL := "CREATE TABLE ClipboardHistory (" .
               "ID INTEGER PRIMARY KEY AUTOINCREMENT, " .
               "Timestamp DATETIME DEFAULT (datetime('now', 'localtime')), " .
               "Content TEXT NOT NULL, " .
               "DataType TEXT, " .
               "SourceApp TEXT, " .
               "SourceTitle TEXT, " .
               "SourcePath TEXT, " .
               "CharCount INTEGER, " .
               "WordCount INTEGER, " .
               "MetaData TEXT, " .
               "IsPinned INTEGER NOT NULL DEFAULT 0)"
        
        if (!ClipboardDB.Exec(SQL)) {
            MsgBox("创建数据库表失败: " . ClipboardDB.ErrorMsg, "数据库初始化错误", "IconX")
            ClipboardDB.CloseDB()
            ClipboardDB := 0
            return
        }
        
        ; 创建索引以提升查询性能
        SQL := "CREATE INDEX IF NOT EXISTS idx_clipboard_timestamp ON ClipboardHistory(Timestamp DESC)"
        ClipboardDB.Exec(SQL)
        
        SQL := "CREATE INDEX IF NOT EXISTS idx_clipboard_content ON ClipboardHistory(Content COLLATE NOCASE)"
        ClipboardDB.Exec(SQL)
        
        SQL := "CREATE INDEX IF NOT EXISTS idx_clipboard_datatype ON ClipboardHistory(DataType COLLATE NOCASE)"
        ClipboardDB.Exec(SQL)
        
        SQL := "CREATE INDEX IF NOT EXISTS idx_clipboard_sourceapp ON ClipboardHistory(SourceApp COLLATE NOCASE)"
        ClipboardDB.Exec(SQL)
        
        SQL := "CREATE INDEX IF NOT EXISTS idx_clipboard_ispinned ON ClipboardHistory(IsPinned)"
        ClipboardDB.Exec(SQL)
        
    } catch as e {
        MsgBox("初始化数据库时发生异常: " . e.Message, "数据库初始化错误", "IconX")
        ClipboardDB := 0
    }
}
```
