    if (hwnd = FloatingToolbarGUI.Hwnd) {
        ; 0xA1 是 WM_NCLBUTTONDOWN, 2 代表标题栏 (Caption)
        ; 这样用户点击长条任何空白处都能像拖动标题栏一样移动它
        PostMessage(0xA1, 2,,, "ahk_id " hwnd)
    }
}

DrawFloatingToolbar() {
    ; ... 原有的 GDI+ 初始化 ...
    
    for index, btn in FloatingToolbarButtons {
        ; 1. 确定当前按钮的颜色状态
        if (index = FloatingToolbarClickedButton)
            currentBrush := BrushClicked  ; 按下状态颜色
        else if (index = FloatingToolbarHoveredButton)
            currentBrush := BrushHover    ; 悬停状态颜色
        else
            currentBrush := BrushNormal   ; 默认颜色
            
        ; 2. 绘制圆角背景
        Gdip_FillRoundedRectangle(G, currentBrush, btn.x, btn.y, btn.w, btn.h, 5)
        
        ; 3. 绘制文字/图标
        Gdip_TextToGraphics(G, btn.text, "c" TextColor " s10 Center vCenter x" btn.x " y" btn.y " w" btn.w " h" btn.h, "Segoe MDL2 Assets")
    }
    
    ; 4. 更新窗口显示
    UpdateLayeredWindow(FloatingToolbarGUI.Hwnd, hdc, ...)
}

    static LastHovered := 0  ; 静态变量，记录上一次悬停状态
    
    if (!FloatingToolbarIsVisible || FloatingToolbarDragging)
        return
        
    ; 获取鼠标相对于屏幕的位置，并转换为相对于窗口的位置
    MouseGetPos(&mx, &my)
    FloatingToolbarGUI.GetPos(&wx, &wy, &ww, &wh)
    
    relX := mx - wx
    relY := my - wy
    
    currentHover := 0
    ; 检测鼠标是否在某个按钮范围内
    for index, btn in FloatingToolbarButtons {
        if (relX >= btn.x && relX <= btn.x + btn.w && relY >= btn.y && relY <= btn.y + btn.h) {
            currentHover := index
            break
        }
    }
    
    ; --- 关键：状态改变时才触发提示 ---
    if (currentHover != LastHovered) {
        if (currentHover > 0) {
            ; 获取当前按钮的提示词
            action := FloatingToolbarButtons[currentHover].action
            tipText := GetButtonTip(action)
            ; 在鼠标附近显示提示
            ToolTip(tipText)
            ; 强制重绘，让按钮显示悬停高亮色
            FloatingToolbarHoveredButton := currentHover
            DrawFloatingToolbar()
        } else {
            ; 鼠标移出按钮区域，关闭提示
            ToolTip()
            FloatingToolbarHoveredButton := 0
            DrawFloatingToolbar()
        }
        LastHovered := currentHover
    }
}

3. 运行优化建议（针对小白友好度）
为什么之前的代码“还是不行”？因为分层窗口的特殊性，任何微小的坐标偏移都会导致判断失效。

性能平衡：建议将定时器频率设为 100ms。 SetTimer(OnFloatingToolbarTimer, 100) 频率太高（如 10ms）会占 CPU，太低则会让小白觉得“提示弹出不灵敏”。

视觉反馈一致性：在 currentHover != LastHovered 发生时，必须调用 DrawFloatingToolbar()。这样用户在看到提示文字的同时，图标背景也会变色。这种**“视觉+文本”的双重确认**对新手极其重要。

防干扰处理：在用户拖动窗口期间（FloatingToolbarDragging = true），应该强制执行 ToolTip()。防止拖动时提示文字像“幽灵”一样粘在屏幕上甩不掉。
