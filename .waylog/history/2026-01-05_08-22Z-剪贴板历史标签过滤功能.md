; ===================== 剪贴板变化处理器 =====================
ProcessClipboardChange(Type) {
    global LastClipboardContent
    
    try {
        ; --- 1. 基础过滤系统 (所有用户) ---
        if (Type = 1) { ; 文本
            content := A_Clipboard
            trimmedContent := Trim(content)
            
            if (StrLen(trimmedContent) < 3) return ; 过滤过短字符
            if (RegExMatch(trimmedContent, "^[\s\r\n]*$")) return ; 过滤纯空白
            
            ; --- 2. 专业过滤系统 (针对 Cursor/开发者) ---
            ; 过滤 node_modules, .git 等开发干扰路径
            if (RegExMatch(content, "i)(node_modules|\.git|__pycache__|\.vs|obj|bin)")) return
            ; 过滤 API Keys (如 OpenAI)
            if (RegExMatch(content, "sk-[a-zA-Z0-9]{32,}")) return
            ; 过滤 Base64 图片字符串
            if (InStr(content, "data:image/") && InStr(content, "base64")) return

            ; --- 3. 类型识别打标 ---
            detectedType := "Text"
            ; 识别十六进制色值
            if (RegExMatch(trimmedContent, "^#([A-Fa-f0-9]{6}|[A-Fa-f0-9]{3})$")) {
                detectedType := "Color"
            }
            ; 识别链接
            else if (RegExMatch(trimmedContent, "i)^https?://\S+")) {
                detectedType := "Link"
            }
            ; 识别代码特征 (简单判断)
            else if (RegExMatch(content, "[\{\}\[\];]")) {
                detectedType := "Code"
            }

            ; 去重检查
            if (content = LastClipboardContent) return
            
            ; 执行保存
            SaveToClipboardFTS5(content, GetActiveApp(), detectedType)
        }
        else if (Type = 2) { ; 图片
            CaptureClipboardImageToFTS5(GetActiveApp())
        }
    }
}

GetActiveApp() {
    try {
        activeHwnd := WinExist("A")
        return WinGetProcessName(activeHwnd)
    } catch {
        return "Unknown"
    }
}

OnNotify_CustomDraw(wParam, lParam, msg, hwnd) {
    static NM_CUSTOMDRAW := -12
    static CDDS_PREPAINT := 0x1
    static CDDS_ITEMPREPAINT := 0x10001
    static CDRF_NOTIFYITEMDRAW := 0x20
    static CDRF_SKIPDEFAULT := 0x4

    code := NumGet(lParam, A_PtrSize * 2, "Int")
    if (code != NM_CUSTOMDRAW) return

    drawStage := NumGet(lParam, A_PtrSize * 3, "UInt")
    if (drawStage == CDDS_PREPAINT) return CDRF_NOTIFYITEMDRAW

    if (drawStage == CDDS_ITEMPREPAINT) {
        rowIdx := NumGet(lParam, A_PtrSize * 5, "Ptr") + 1
        if (rowIdx <= HistoryDisplayCache.Length) {
            rowData := HistoryDisplayCache[rowIdx]
            if (rowData.Has("DataType") && rowData["DataType"] = "Color") {
                hdc := NumGet(lParam, A_PtrSize * 4, "Ptr")
                ; 绘制色块矩形
                DrawColorBox(hdc, rowIdx, rowData["Content"])
                return CDRF_SKIPDEFAULT ; 拦截默认绘制
            }
        }
    }
}

DrawColorBox(hdc, rowIdx, hexColor) {
    ; 计算单元格矩形 (简化逻辑)
    rect := Buffer(16, 0)
    NumPut("Int", 2, rect, 4) ; LVIR_BOUNDS
    SendMessage(0x1038, rowIdx - 1, rect.Ptr, HistoryListView.Hwnd)
    
    left := NumGet(rect, 0, "Int") + 10
    top := NumGet(rect, 4, "Int") + 6
    
    ; 绘制背景小方块
    bgr := ConvertHexToBGR(hexColor)
    hBrush := DllCall("CreateSolidBrush", "UInt", bgr, "Ptr")
    box := Buffer(16)
    NumPut("Int", left, box, 0), NumPut("Int", top, box, 4)
    NumPut("Int", left+12, box, 8), NumPut("Int", top+12, box, 12)
    DllCall("FillRect", "Ptr", hdc, "Ptr", box.Ptr, "Ptr", hBrush)
    DllCall("DeleteObject", "Ptr", hBrush)
}

第一步：升级数据库表结构（底层移植）
目前的 clipboard.ahk 可能使用的是普通 SQLite 表。要获得 FTS5 的性能，必须将 ClipboardHistory 表声明为 VIRTUAL TABLE 并使用 fts5 引擎。

修改 InitClipboardFTS5DB 函数：

代码段

InitClipboardFTS5DB() {
    ; 关键点：使用 VIRTUAL TABLE 和 fts5 引擎
    ; 使用 unicode61 分词器以完美支持中文搜索
    sql := "
    (
        CREATE VIRTUAL TABLE IF NOT EXISTS ClipboardHistory USING fts5(
            Content, 
            SourceApp, 
            DataType, 
            LastCopyTime UNINDEXED, 
            SourcePath UNINDEXED,
            tokenize='unicode61'
        );
    )"
    
    ; 调用你的数据库执行逻辑
    if (ClipboardFTS5DB.Execute(sql)) {
        RecordClipboardEvent("数据库", "FTS5 虚拟表初始化成功")
    }
}
注：UNINDEXED 表示该列不参与全文索引，仅用于存储，这样可以节省空间并提升保存速度。

RefreshHistoryData() {
    global HistorySearchEdit, TagFilter
    searchText := HistorySearchEdit.Value
    
    ; 1. 基础 SQL
    sql := "SELECT * FROM ClipboardHistory"
    
    ; 2. FTS5 全文搜索逻辑 (核心差异点)
    if (searchText != "") {
        ; 使用 MATCH 语法代替 LIKE，性能提升百倍
        ; 添加 * 号实现前缀匹配（输入 "func" 也能匹配到 "function"）
        sql .= " WHERE Content MATCH '" . searchText . "*'"

    if (TagFilter != "全部") {
        ; 如果已经有了 WHERE 关键词，则用 AND
        prefix := InStr(sql, "WHERE") ? " AND " : " WHERE "
        sql .= prefix . "DataType = '" . TagFilter . "'"
    }
    
    sql .= " ORDER BY LastCopyTime DESC LIMIT 100"

}

global HistorySearchTimer := 0

OnSearchEditChange(*) {
    global HistorySearchTimer
    
    ; 如果用户正在快速输入，取消上一个还没执行的刷新任务
