getBestIconPath(item) {
    ; 1. 检查是否是真实文件/文件夹
    if (FileExist(item.Content))
        return item.Content
    
    ; 2. 根据 DataType 分支
    switch item.DataType {
        case "Image": return item.ImagePath || ".png"
        case "Link":  return ".url"
        case "Code":  return item.Extension || ".txt"
        case "Color": return "shell32.dll,245" ; 举例：借用系统的调色板图标
    }
    
    ; 3. 兜底：来源应用路径
    if (item.SourcePath != "")
        return item.SourcePath
        
    return ".txt" ; 最终兜底
}
